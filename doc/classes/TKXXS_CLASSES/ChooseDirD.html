  hash = {
    :mode => :choosedir,

    :title=>"Choose Directory",

def choose( hash )
  dirStr = Tk.chooseDirectory( hash )
  path =  dirStr.empty?  ?  nil  :  dirStr

  if path
    set_paths( path )
  else
    validate_and_leave
  end
  nil
end
